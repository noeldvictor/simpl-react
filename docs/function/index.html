    public simpl(options: Object): * source 

    import {simpl} from 'simpl/lib/decorators/simpl'

    Decorator to wrap your main app

    Params:

    NameTypeAttributeDescription
    options Object

    An object of options.
    options.username string

    The username for authenticating on the WAMP Router.
    options.password string

    The password for authenticating on the WAMP Router.
    options.url string

    The URL of the WAMP router.
    options.progressComponent function
    • optional

    A customized Component to show the App's connection state. The component will receive a progress prop which can have one of the following values:

    • offline: The initial value. The app is not connected to the WAMP Router, yet.
    • connected: The app is connected and authenticated, but it still needs to download data.
    • loaded: The app has downloaded all the relevant data.
    options.topics function

    A function return a list of topic to subsribe to.
    options.prefixes Object
    • optional

    An object mapping names to topic prefixes, to be used as shortcuts.

    Return:

    *

    Example:

    export default simpl({
  username: 'username',
  password: 'password',
  url: 'ws://example.com/ws',
  progressComponent: MyProgressComponent,
  topics: () => ['topic1', 'topic2'],
  prefixes: {
    model: 'org.example.namespace'
  }
})(MyComponent);

    public validateField(options: ValidationOptions): ReactElement source 

    import {validateField} from 'simpl/lib/decorators/forms/validates'

    Add value validation to a single field.

    Params:

    NameTypeAttributeDescription
    options ValidationOptions

    An object of options

    Return:

    ReactElement

    A React component.

    Example:

    import React from 'react';

import {validates} from 'simpl/lib/decorators/forms/validates';
import {FormControl} from 'react-bootstrap';

const noSpam = (value, ownProps) => {
    // validation logic goes here
    if (value.endsWith('@spam.com')) {
        return "that's spam"
    }
};

const itsNew = (value, ownProps) => {
    if (value === ownProps.previousEmail) {
        return "email didn't change"
    }
};

function EmailField(props) {
    // all aestethics goes here
    return (
        <FormControl
            type="email"
            {...props}
        />
    )
}

const ValidatingEmailField = validateField({
    errors: ['isEmail', noSpam],
    warnings: [itsNew]
)(EmailField);

class MyComponent extends React.Component {
    render() {
        return(
            <div>
                <ValidatingEmailField
                    previousEmail="my@email.com"
                    errors={['isRequired']}
                />
            </div>
        );
    }
}
